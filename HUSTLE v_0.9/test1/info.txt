CREATE TABLE course (
    id int,--course的编号
    name VARCHAR(20),--course的名称
    start int,--course开始的周（1-17）
    end int,--course结束的周（1-17，为了简单就默认连续吧= =）
    first int,--在星期几（1-7）
    second int,--在星期几（0-7,0代表没有）
    time int--course在第几节（1-11，默认占两节吧！）
);
CREATE TABLE info(
    user_id int, -- 学生及教师编号
    course_id int, -- 课程号
    first_attendance int,
    second_attendance int,
    exp int,
    homework int
);
